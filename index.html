	scanf("%d%d",&n,&k);
	m=0;
	for (int i=2;i*i<=k;i++)
		if (k%i==0){

		int x=q[i].fi,tmp=0;

		res=min(res,tmp/q[i].se);
	}
	printf("%d\n",res);
	return 0;
}
